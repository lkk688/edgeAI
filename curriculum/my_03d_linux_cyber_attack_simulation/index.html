    # iot_scanner.py
import nmap
import requests
import json
from scapy.all import *

class IoTDeviceScanner:
    def __init__(self, target_range):
        self.target_range = target_range
        self.nm = nmap.PortScanner()
        self.devices = []

    def scan_network(self):
        """Scan for IoT devices using various techniques"""
        # ARP scan for live hosts
        arp_scan = self.nm.scan(hosts=self.target_range, arguments='-sn')

        for host in self.nm.all_hosts():
            device_info = self.fingerprint_device(host)
            if device_info:
                self.devices.append(device_info)

    def fingerprint_device(self, host):
        """Fingerprint IoT device characteristics"""
        device = {'ip': host, 'services': [], 'vulnerabilities': []}

        # Port scan
        self.nm.scan(host, '1-65535', '-sV -sC')

        if host in self.nm.all_hosts():
            for proto in self.nm[host].all_protocols():
                ports = self.nm[host][proto].keys()
                for port in ports:
                    service = self.nm[host][proto][port]
                    device['services'].append({
                        'port': port,
                        'service': service['name'],
                        'version': service.get('version', 'unknown')
                    })

        # Check for common IoT vulnerabilities
        self.check_iot_vulns(device)
        return device

    def check_iot_vulns(self, device):
        """Check for common IoT vulnerabilities"""
        # Default credentials check
        default_creds = [
            ('admin', 'admin'), ('admin', 'password'),
            ('root', 'root'), ('admin', '12345')
        ]

        for service in device['services']:
            if service['service'] in ['ssh', 'telnet', 'http']:
                for user, pwd in default_creds:
                    if self.test_credentials(device['ip'], service['port'], user, pwd):
                        device['vulnerabilities'].append({
                            'type': 'default_credentials',
                            'service': service['service'],
                            'credentials': f"{user}:{pwd}"
                        })

    def test_credentials(self, host, port, username, password):
        """Test credentials against service"""
        try:
            if port == 22:  # SSH
                import paramiko
                ssh = paramiko.SSHClient()
                ssh.set_missing_host_key_policy(paramiko.AutoAddPolicy())
                ssh.connect(host, port, username, password, timeout=5)
                ssh.close()
                return True
        except:
            pass
        return False

# Usage
scanner = IoTDeviceScanner('172.20.0.0/16')
scanner.scan_network()
print(json.dumps(scanner.devices, indent=2))

    # security_monitor.py
import psutil
import subprocess
import json
import time
import threading
from datetime import datetime
import re
from collections import defaultdict

class JetsonSecurityMonitor:
    def __init__(self):
        self.alerts = []
        self.network_baseline = {}
        self.process_baseline = set()
        self.monitoring = False
        self.threat_indicators = {
            'port_scan': re.compile(r'SYN.*flags.*'),
            'brute_force': re.compile(r'Failed password.*'),
            'reverse_shell': re.compile(r'nc.*-e.*|bash.*-i.*'),
            'privilege_escalation': re.compile(r'sudo.*COMMAND.*')
        }

    def start_monitoring(self):
        """Start comprehensive security monitoring"""
        self.monitoring = True

        # Start monitoring threads
        threads = [
            threading.Thread(target=self.monitor_network_traffic),
            threading.Thread(target=self.monitor_system_processes),
            threading.Thread(target=self.monitor_file_integrity),
            threading.Thread(target=self.monitor_authentication_logs),
            threading.Thread(target=self.analyze_behavioral_patterns)
        ]

        for thread in threads:
            thread.daemon = True
            thread.start()

        print("[+] Security monitoring started")

    def monitor_network_traffic(self):
        """Monitor network traffic for suspicious patterns"""
        while self.monitoring:
            try:
                # Capture network packets
                result = subprocess.run([
                    'tcpdump', '-i', 'any', '-c', '100', '-n'
                ], capture_output=True, text=True, timeout=30)

                self.analyze_network_patterns(result.stdout)
                time.sleep(10)
            except Exception as e:
                print(f"Network monitoring error: {e}")

    def analyze_network_patterns(self, traffic_data):
        """Analyze network traffic for attack patterns"""
        lines = traffic_data.split('\n')

        # Detect port scanning
        syn_packets = defaultdict(int)
        for line in lines:
            if 'Flags [S]' in line:  # SYN packets
                src_ip = self.extract_src_ip(line)
                syn_packets[src_ip] += 1

        # Alert on potential port scan
        for ip, count in syn_packets.items():
            if count > 20:  # Threshold for port scan
                self.create_alert('port_scan', f"Potential port scan from {ip}: {count} SYN packets")

    def monitor_system_processes(self):
        """Monitor system processes for suspicious activity"""
        while self.monitoring:
            try:
                current_processes = set()
                for proc in psutil.process_iter(['pid', 'name', 'cmdline']):
                    try:
                        cmdline = ' '.join(proc.info['cmdline'] or [])
                        current_processes.add((proc.info['name'], cmdline))

                        # Check for suspicious commands
                        self.check_suspicious_process(proc.info)
                    except (psutil.NoSuchProcess, psutil.AccessDenied):
                        continue

                # Detect new processes
                new_processes = current_processes - self.process_baseline
                if new_processes and self.process_baseline:
                    for name, cmdline in new_processes:
                        self.create_alert('new_process', f"New process detected: {name} - {cmdline}")

                self.process_baseline = current_processes
                time.sleep(5)
            except Exception as e:
                print(f"Process monitoring error: {e}")

    def check_suspicious_process(self, proc_info):
        """Check if process exhibits suspicious behavior"""
        cmdline = ' '.join(proc_info['cmdline'] or [])

        suspicious_patterns = [
            r'nc.*-e.*',  # Netcat reverse shell
            r'bash.*-i.*',  # Interactive bash
            r'/tmp/.*',  # Execution from /tmp
            r'wget.*\|.*sh',  # Download and execute
            r'curl.*\|.*sh',  # Download and execute
            r'python.*-c.*socket',  # Python reverse shell
        ]

        for pattern in suspicious_patterns:
            if re.search(pattern, cmdline, re.IGNORECASE):
                self.create_alert('suspicious_process', 
                    f"Suspicious process: {proc_info['name']} - {cmdline}")

    def monitor_authentication_logs(self):
        """Monitor authentication logs for brute force attempts"""
        failed_attempts = defaultdict(int)

        while self.monitoring:
            try:
                # Read auth logs
                result = subprocess.run([
                    'tail', '-n', '50', '/var/log/auth.log'
                ], capture_output=True, text=True)

                for line in result.stdout.split('\n'):
                    if 'Failed password' in line:
                        ip = self.extract_ip_from_log(line)
                        if ip:
                            failed_attempts[ip] += 1

                            if failed_attempts[ip] > 5:
                                self.create_alert('brute_force', 
                                    f"Brute force attack detected from {ip}: {failed_attempts[ip]} attempts")
                                self.block_ip(ip)

                time.sleep(30)
            except Exception as e:
                print(f"Auth log monitoring error: {e}")

    def monitor_file_integrity(self):
        """Monitor critical files for unauthorized changes"""
        critical_files = [
            '/etc/passwd', '/etc/shadow', '/etc/sudoers',
            '/etc/ssh/sshd_config', '/etc/crontab'
        ]

        file_hashes = {}

        # Initial hash calculation
        for file_path in critical_files:
            try:
                result = subprocess.run(['sha256sum', file_path], 
                    capture_output=True, text=True)
                file_hashes[file_path] = result.stdout.split()[0]
            except:
                continue

        while self.monitoring:
            try:
                for file_path in critical_files:
                    result = subprocess.run(['sha256sum', file_path], 
                        capture_output=True, text=True)
                    current_hash = result.stdout.split()[0]

                    if file_path in file_hashes and file_hashes[file_path] != current_hash:
                        self.create_alert('file_integrity', 
                            f"Critical file modified: {file_path}")
                        file_hashes[file_path] = current_hash

                time.sleep(60)
            except Exception as e:
                print(f"File integrity monitoring error: {e}")

    def analyze_behavioral_patterns(self):
        """Analyze system behavior for anomalies"""
        while self.monitoring:
            try:
                # CPU usage analysis
                cpu_percent = psutil.cpu_percent(interval=1)
                if cpu_percent > 90:
                    self.create_alert('resource_anomaly', 
                        f"High CPU usage detected: {cpu_percent}%")

                # Memory usage analysis
                memory = psutil.virtual_memory()
                if memory.percent > 90:
                    self.create_alert('resource_anomaly', 
                        f"High memory usage detected: {memory.percent}%")

                # Network connections analysis
                connections = psutil.net_connections()
                external_connections = [conn for conn in connections 
                    if conn.raddr and not conn.raddr.ip.startswith('127.')]

                if len(external_connections) > 50:
                    self.create_alert('network_anomaly', 
                        f"High number of external connections: {len(external_connections)}")

                time.sleep(30)
            except Exception as e:
                print(f"Behavioral analysis error: {e}")

    def create_alert(self, alert_type, message):
        """Create and log security alert"""
        alert = {
            'timestamp': datetime.now().isoformat(),
            'type': alert_type,
            'message': message,
            'severity': self.get_severity(alert_type)
        }

        self.alerts.append(alert)
        print(f"[ALERT] {alert['timestamp']} - {alert_type.upper()}: {message}")

        # Auto-response for critical alerts
        if alert['severity'] == 'critical':
            self.auto_respond(alert)

    def get_severity(self, alert_type):
        """Determine alert severity"""
        severity_map = {
            'port_scan': 'medium',
            'brute_force': 'high',
            'suspicious_process': 'high',
            'file_integrity': 'critical',
            'resource_anomaly': 'medium',
            'network_anomaly': 'medium'
        }
        return severity_map.get(alert_type, 'low')

    def auto_respond(self, alert):
        """Automated response to critical alerts"""
        if alert['type'] == 'brute_force':
            ip = self.extract_ip_from_message(alert['message'])
            if ip:
                self.block_ip(ip)
        elif alert['type'] == 'suspicious_process':
            # Could implement process termination here
            pass

    def block_ip(self, ip):
        """Block IP address using iptables"""
        try:
            subprocess.run(['iptables', '-A', 'INPUT', '-s', ip, '-j', 'DROP'])
            print(f"[+] Blocked IP: {ip}")
        except Exception as e:
            print(f"Failed to block IP {ip}: {e}")

    def extract_src_ip(self, line):
        """Extract source IP from network traffic line"""
        match = re.search(r'(\d+\.\d+\.\d+\.\d+)', line)
        return match.group(1) if match else None

    def extract_ip_from_log(self, line):
        """Extract IP from log line"""
        match = re.search(r'from (\d+\.\d+\.\d+\.\d+)', line)
        return match.group(1) if match else None

    def extract_ip_from_message(self, message):
        """Extract IP from alert message"""
        match = re.search(r'(\d+\.\d+\.\d+\.\d+)', message)
        return match.group(1) if match else None

    def get_security_report(self):
        """Generate security report"""
        report = {
            'total_alerts': len(self.alerts),
            'alerts_by_type': defaultdict(int),
            'alerts_by_severity': defaultdict(int),
            'recent_alerts': self.alerts[-10:]
        }

        for alert in self.alerts:
            report['alerts_by_type'][alert['type']] += 1
            report['alerts_by_severity'][alert['severity']] += 1

        return report

# Usage
monitor = JetsonSecurityMonitor()
monitor.start_monitoring()

# Keep monitoring running
try:
    while True:
        time.sleep(60)
        report = monitor.get_security_report()
        print(f"\n[REPORT] Total alerts: {report['total_alerts']}")
except KeyboardInterrupt:
    monitor.monitoring = False
    print("\n[+] Security monitoring stopped")

    # advanced_firewall.py
import subprocess
import json
from datetime import datetime, timedelta

class JetsonFirewall:
    def __init__(self):
        self.blocked_ips = set()
        self.rules = []
        self.whitelist = {'127.0.0.1', '172.20.0.20'}  # Defender IP

    def setup_base_rules(self):
        """Setup base firewall rules"""
        base_rules = [
            # Allow loopback
            'iptables -A INPUT -i lo -j ACCEPT',
            # Allow established connections
            'iptables -A INPUT -m state --state ESTABLISHED,RELATED -j ACCEPT',
            # Allow SSH from specific subnet only
            'iptables -A INPUT -p tcp --dport 22 -s 172.20.0.0/16 -j ACCEPT',
            # Rate limit SSH connections
            'iptables -A INPUT -p tcp --dport 22 -m limit --limit 3/min -j ACCEPT',
            # Drop all other SSH
            'iptables -A INPUT -p tcp --dport 22 -j DROP',
            # Allow HTTP/HTTPS
            'iptables -A INPUT -p tcp --dport 80 -j ACCEPT',
            'iptables -A INPUT -p tcp --dport 443 -j ACCEPT',
            # Log dropped packets
            'iptables -A INPUT -j LOG --log-prefix "DROPPED: "',
            # Default drop
            'iptables -A INPUT -j DROP'
        ]

        for rule in base_rules:
            self.execute_rule(rule)

    def block_ip_temporary(self, ip, duration_minutes=60):
        """Block IP temporarily"""
        if ip in self.whitelist:
            print(f"[!] IP {ip} is whitelisted, not blocking")
            return

        # Add to iptables
        rule = f'iptables -I INPUT -s {ip} -j DROP'
        self.execute_rule(rule)

        self.blocked_ips.add(ip)
        print(f"[+] Blocked {ip} for {duration_minutes} minutes")

        # Schedule unblock (in real implementation, use proper scheduler)
        # For demo, we'll just track it
        unblock_time = datetime.now() + timedelta(minutes=duration_minutes)
        print(f"[+] {ip} will be unblocked at {unblock_time}")

    def execute_rule(self, rule):
        """Execute iptables rule"""
        try:
            subprocess.run(rule.split(), check=True)
            self.rules.append(rule)
            print(f"[+] Executed: {rule}")
        except subprocess.CalledProcessError as e:
            print(f"[-] Failed to execute: {rule} - {e}")

    def get_blocked_ips(self):
        """Get list of currently blocked IPs"""
        return list(self.blocked_ips)

    def unblock_ip(self, ip):
        """Unblock IP address"""
        rule = f'iptables -D INPUT -s {ip} -j DROP'
        self.execute_rule(rule)
        self.blocked_ips.discard(ip)
        print(f"[+] Unblocked {ip}")

# Usage
firewall = JetsonFirewall()
firewall.setup_base_rules()

    def start_dashboard(self):
        """Start the curses-based security dashboard"""
        curses.wrapper(self.main_loop)

    def main_loop(self, stdscr):
        """Main dashboard loop"""
        self.running = True
        curses.curs_set(0)  # Hide cursor
        stdscr.nodelay(1)   # Non-blocking input

        while self.running:
            stdscr.clear()

            # Header
            stdscr.addstr(0, 0, "üõ°Ô∏è  JETSON SECURITY DASHBOARD", curses.A_BOLD)
            stdscr.addstr(1, 0, "=" * 60)

            # System status
            self.draw_system_status(stdscr, 3)

            # Recent alerts
            self.draw_recent_alerts(stdscr, 10)

            # Network activity
            self.draw_network_activity(stdscr, 20)

            # Controls
            stdscr.addstr(35, 0, "Press 'q' to quit, 'r' to refresh")

            stdscr.refresh()

            # Check for input
            key = stdscr.getch()
            if key == ord('q'):
                self.running = False

            time.sleep(1)

    def draw_system_status(self, stdscr, start_row):
        """Draw system status section"""
        stdscr.addstr(start_row, 0, "üìä SYSTEM STATUS:", curses.A_BOLD)

        import psutil
        cpu_percent = psutil.cpu_percent()
        memory = psutil.virtual_memory()

        stdscr.addstr(start_row + 1, 2, f"CPU Usage: {cpu_percent:.1f}%")
        stdscr.addstr(start_row + 2, 2, f"Memory Usage: {memory.percent:.1f}%")
        stdscr.addstr(start_row + 3, 2, f"Active Connections: {len(psutil.net_connections())}")
        stdscr.addstr(start_row + 4, 2, f"Monitoring Status: {'üü¢ ACTIVE' if self.monitor.monitoring else 'üî¥ INACTIVE'}")

    def draw_recent_alerts(self, stdscr, start_row):
        """Draw recent alerts section"""
        stdscr.addstr(start_row, 0, "üö® RECENT ALERTS:", curses.A_BOLD)

        recent_alerts = self.monitor.alerts[-8:] if self.monitor.alerts else []

        for i, alert in enumerate(recent_alerts):
            severity_color = curses.A_NORMAL
            if alert['severity'] == 'critical':
                severity_color = curses.A_REVERSE
            elif alert['severity'] == 'high':
                severity_color = curses.A_BOLD

            alert_text = f"{alert['timestamp'][:19]} - {alert['type'].upper()}: {alert['message'][:40]}..."
            stdscr.addstr(start_row + 1 + i, 2, alert_text, severity_color)

    def draw_network_activity(self, stdscr, start_row):
        """Draw network activity section"""
        stdscr.addstr(start_row, 0, "üåê NETWORK ACTIVITY:", curses.A_BOLD)

        # This would show real network stats
        stdscr.addstr(start_row + 1, 2, "Active monitoring of network traffic...")
        stdscr.addstr(start_row + 2, 2, f"Packets analyzed: {len(self.monitor.alerts) * 100}")
        stdscr.addstr(start_row + 3, 2, f"Threats detected: {len([a for a in self.monitor.alerts if a['severity'] in ['high', 'critical']])}")
